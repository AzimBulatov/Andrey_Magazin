Листинг 4 – Файл docker-compose.yml (Конфигурация инфраструктуры)

services:
  postgres:
    image: postgres:15-alpine
    container_name: telegram-shop-db
    restart: always
    environment:
      POSTGRES_USER: postgres
      POSTGRES_PASSWORD: ${POSTGRES_PASSWORD:-postgres}
      POSTGRES_DB: telegram_shop
    ports:
      - "5432:5432"
    volumes:
      - postgres_data:/var/lib/postgresql/data
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U postgres"]
      interval: 5s
      timeout: 5s
      retries: 5
    networks:
      - app-network

  redis:
    image: redis:7-alpine
    container_name: telegram-shop-redis
    restart: always
    ports:
      - "6379:6379"
    networks:
      - app-network

  backend:
    build:
      context: ./backend
      dockerfile: Dockerfile
    container_name: telegram-shop-backend
    restart: always
    ports:
      - "3000:3000"
    environment:
      DATABASE_URL: postgresql://postgres:${POSTGRES_PASSWORD:-postgres}@postgres:5432/telegram_shop
      JWT_SECRET: ${JWT_SECRET:-dev-secret-key}
      TELEGRAM_BOT_TOKEN: ${TELEGRAM_BOT_TOKEN}
      PORT: 3000
      API_URL: ${API_URL:-http://localhost:3000}
      WEBSITE_URL: ${WEBSITE_URL:-http://localhost:3002}
      WEBHOOK_DOMAIN: ${WEBHOOK_DOMAIN:-}
      YOOKASSA_SHOP_ID: ${YOOKASSA_SHOP_ID}
      YOOKASSA_SECRET_KEY: ${YOOKASSA_SECRET_KEY}
      REDIS_HOST: redis
      REDIS_PORT: 6379
    volumes:
      - uploads_data:/app/uploads
    depends_on:
      postgres:
        condition: service_healthy
      redis:
        condition: service_started
    networks:
      - app-network

  client-site:
    build:
      context: ./client-site
      dockerfile: Dockerfile
    container_name: telegram-shop-client
    restart: always
    networks:
      - app-network

  nginx:
    image: nginx:alpine
    container_name: telegram-shop-nginx
    restart: always
    ports:
      - "80:80"
    volumes:
      - ./nginx/nginx.conf:/etc/nginx/nginx.conf:ro
      - ./nginx/conf.d:/etc/nginx/conf.d:ro
      - uploads_data:/var/www/uploads:ro
    depends_on:
      - backend
      - client-site
    networks:
      - app-network

volumes:
  postgres_data:
  uploads_data:

networks:
  app-network:
    driver: bridge
